<script>
const totalParticipants = 200;
const awards = {"一等奖":3,"二等奖":5,"三等奖":10};
const participationCount = totalParticipants - Object.values(awards).reduce((a,b)=>a+b,0);
awards["参与奖"] = participationCount;

let prizePool = [];
for(const [award,count] of Object.entries(awards)){
  for(let i=0;i<count;i++) prizePool.push(award);
}
prizePool = prizePool.sort(()=>Math.random()-0.5);

const awardColors = {"一等奖":"#FFD700","二等奖":"#C0C0C0","三等奖":"#CD7F32","参与奖":"#90EE90"};
let records=[];

const canvas = document.getElementById("wheel");
const ctx = canvas.getContext("2d");
const centerX = canvas.width/2;
const centerY = canvas.height/2;
const radius = 180;

function drawWheel(){
  const len = prizePool.length;
  const anglePer = 2*Math.PI/len;
  for(let i=0;i<len;i++){
    const start = i*anglePer;
    const end = start + anglePer;
    ctx.beginPath();
    ctx.moveTo(centerX, centerY);
    const prize = prizePool[i];
    ctx.fillStyle = awardColors[prize] || "#fff";
    ctx.arc(centerX, centerY, radius, start, end);
    ctx.closePath();
    ctx.fill();

    ctx.save();
    ctx.translate(centerX, centerY);
    ctx.rotate(start + anglePer/2);
    ctx.fillStyle = "#000";
    ctx.font = "16px Microsoft YaHei";
    ctx.textAlign = "right";
    ctx.fillText(prize, radius-10, 5);
    ctx.restore();
  }
}

drawWheel();

let spinning = false;
let currentAngle = 0;

function startDraw(){
  if(spinning) return;

  const name = document.getElementById("name").value.trim();
  const unit = document.getElementById("unit").value.trim();
  if(!name||!unit){
    alert("请填写姓名和单位！");
    return;
  }
  if(prizePool.length===0){
    alert("奖池已空！");
    return;
  }

  spinning = true;
  const spinAudio = document.getElementById("spinSound");
  spinAudio.currentTime = 0;
  spinAudio.play();

  const targetIndex = Math.floor(Math.random()*prizePool.length);
  const totalAngle = 10*2*Math.PI + (targetIndex*(2*Math.PI/prizePool.length));
  const duration = 4000;
  const startTime = performance.now();

  function animate(now){
    const elapsed = now-startTime;
    const progress = Math.min(elapsed/duration,1);
    const easeOut = 1 - Math.pow(1-progress,3);
    currentAngle = easeOut*totalAngle;
    ctx.clearRect(0,0,canvas.width,canvas.height);
    ctx.save();
    ctx.translate(centerX,centerY);
    ctx.rotate(currentAngle);
    ctx.translate(-centerX,-centerY);
    drawWheel();
    ctx.restore();

    if(progress<1){
      requestAnimationFrame(animate);
    } else {
      spinning=false;
      spinAudio.pause();
      const prize = prizePool.splice(targetIndex,1)[0];
      records.push({name,unit,prize});
      document.getElementById("result").innerText = `${name}，恭喜您抽中了：${prize}！`;
      document.getElementById("winSound").play();
      updateRecordTable();
    }
  }

  requestAnimationFrame(animate);
}

function updateRecordTable(){
  const tbody = document.getElementById("recordTable").getElementsByTagName("tbody")[0];
  tbody.innerHTML = "";
  records.forEach((r,index)=>{
    const row = tbody.insertRow();
    row.insertCell(0).innerText = index+1;
    row.insertCell(1).innerText = r.name;
    row.insertCell(2).innerText = r.unit;
    row.insertCell(3).innerText = r.prize;
  });
}

function saveRecords(){
  if(records.length===0){
    alert("没有抽奖记录！");
    return;
  }
  let csvContent="姓名,单位,奖项\n";
  records.forEach(r=>{csvContent+=`${r.name},${r.unit},${r.prize}\n`;});
  const blob = new Blob([csvContent], {type:"text/csv;charset=utf-8;"});
  const link = document.createElement("a");
  link.href = URL.createObjectURL(blob);
  link.download = "抽奖记录.csv";
  link.click();
}
